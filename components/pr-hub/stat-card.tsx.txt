import * as React from "react";
import { Card } from "@/components/ui/card";
import { cn } from "@/lib/cn";

/**
 * StatCard — the dashboard hierarchy unit.
 * Big brand numeral (.num), mono kicker label, optional delta trend.
 * Trend uses success/danger tone only — no color invented outside the system.
 */
export interface StatCardProps {
  label: string;
  value: string | number;
  unit?: string;
  delta?: { value: string; direction: "up" | "down" | "flat" };
  className?: string;
}

export function StatCard({ label, value, unit, delta, className }: StatCardProps) {
  const trendColor =
    delta?.direction === "up"
      ? "text-success"
      : delta?.direction === "down"
      ? "text-destructive"
      : "text-muted-foreground";
  const arrow = delta?.direction === "up" ? "↑" : delta?.direction === "down" ? "↓" : "→";

  return (
    <Card className={cn("flex flex-col gap-2 p-4", className)}>
      <span className="font-mono text-[11px] uppercase tracking-[0.12em] text-muted-foreground">
        {label}
      </span>
      <div className="flex items-baseline gap-1.5">
        <span className="num text-4xl">{value}</span>
        {unit && <span className="text-sm text-muted-foreground">{unit}</span>}
      </div>
      {delta && (
        <span className={cn("text-xs font-medium", trendColor)}>
          {arrow} {delta.value}
        </span>
      )}
    </Card>
  );
}
